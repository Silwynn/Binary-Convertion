def binary_to_decimal(binary):
    return int(binary, 2)

def decimal_to_hex_oct(decimal):
    return hex(decimal).lstrip("0x").upper(), oct(decimal).lstrip("0o")

def binary_operations_menu():
    choice = int(input("\nBinary Operations Menu:\n[1] Division\n[2] Multiplication\n[3] Subtraction\n[4] Addition\n[5] Negative (2's Complement)\nEnter your choice (1-5): "))
    binary_input = input("Enter the binary number: ").replace(" ", "")

    if choice == 5:
        negative_binary = bin(-int(binary_input, 2) & int('1' * len(binary_input), 2))[2:]
        print(f"Negative (2's Complement): {negative_binary}")
    else:
        print("Invalid choice. Please enter a valid option.")

def main_menu():
    while True:
        choice = int(input("\nMain Menu:\n[1] Binary Operations\n[2] Number System Conversion\n[3] Exit\nEnter your choice (1-3): "))

        if choice == 1:
